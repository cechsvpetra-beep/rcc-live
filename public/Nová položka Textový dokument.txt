    <label for="team">Tím</label>
    <select id="team"></select>

    <label for="weight">Váha (kg)</label>
    <input type="number" id="weight" step="0.01" placeholder="napr. 12.40">

    <button onclick="saveCatch()">Uložiť úlovok</button>

    <div id="msg"></div>
  </div>

  <script>
    const teamSelect = document.getElementById("team");
    const msg = document.getElementById("msg");

    for(let i=1;i<=50;i++){
      const option = document.createElement("option");
      option.value = i;
      option.textContent = "Tím " + i;
      teamSelect.appendChild(option);
    }

    async function saveCatch(){
      const team = document.getElementById("team").value;
      const weight = document.getElementById("weight").value;

      if(!weight){
        msg.textContent = "Zadaj váhu.";
        msg.style.color = "orange";
        return;
      }

      const url = `/catch?team=${team}&weight=${weight}`;

      try{
        const res = await fetch(url);
        const text = await res.text();

        if(text === "ok"){
          msg.textContent = "Úlovok uložený.";
          msg.style.color = "#2ecc71";
          document.getElementById("weight").value = "";
        }else{
          msg.textContent = "Chyba pri ukladaní.";
          msg.style.color = "red";
        }
      }catch(e){
        msg.textContent = "Server neodpovedá.";
        msg.style.color = "red";
      }
    }
  </script>
